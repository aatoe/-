## 节流

```js
onPageScroll() {
	// 节流
	let that = this;
	debounce(1000);
	function debounce(delay){
		// debugger
		function displayFunc() {
			console.log(that.timer)
			console.log("进入闭包,暂未执行函数")
			if (that.timer) {
				clearTimeout(that.timer);    
	　　		}
			that.timer = setTimeout(() => {
				console.log("进入闭包,执行函数")
				that.initClientRect();
				that.timer = null;
			}, delay)	
		}
		displayFunc();
	} 
	
},
// 总结就是有点坑, timer 一直是undefined,那只能放在data中了, 
// 这里适用于滚动的, 但是频繁点击的话(因为默认是第一次是执行的),可以在函数里面给他赋值,第二次执行的时候判断,或者用防抖.
```
